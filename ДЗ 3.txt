enum engine {
    case on, off
}

enum windows {
    case open, close
}

enum fill_trunk {
    case full, empty
}

struct SportCar {
    let brand: String
    let year: Int
    let trunk: Int
    
    var engine: engine {
        willSet {
            newValue == .on ? print ("двигатель \(brand) включен") : print("двигатель \(brand) выключен")
        }
    }
    
    var windows: windows {
        willSet {
            newValue == .open ? print("окна \(brand) открыты") : print("окна \(brand) закрыты")
        }
    }
    
    
    var fill_trunk: fill_trunk {
        willSet {
            newValue == .full ? print("багажник \(brand) заполнен") : print("багажник \(brand) пустой")
        }
    }
    
    var color: String
    
    mutating func color_change(col:String) {
        switch col {
        case "жёлтый":
            self.color = "жёлтый"
            print("цвет \(brand) был изменён на жёлтый")
        case "белый":
            self.color = "белый"
            print("цвет \(brand) был изменён на белый")
        case "розовый":
            self.color = "розовый"
            print("цвет \(brand) был изменён на розовый")
        case "чёрный":
            self.color = "чёрный"
            print("цвет \(brand) был изменён на чёрный")
        default:
            print("Такого цвета нет")
        }
    }
}

var car1 = SportCar (brand: "Acura", year: 2016, trunk: 1000,engine: .on, windows: .open,fill_trunk: .empty, color: "white")
print("\(car1.brand) \(car1.year), вместительность багажника: \(car1.trunk), двигатель \(car1.engine), окна \(car1.windows), багажник \(car1.fill_trunk)")
print()
